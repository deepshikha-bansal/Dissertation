int convert(float num1)
{
     int c;
     float b;
     
     //Scale floating point variable num1 up by a factor 2^frac_len
     b=num1*(2^frac_len);
     
     //use the Round function in the math library of C++ 
     //to round the variable to nearest integer.
     c=roundf(b);
     
     return(c);
}
