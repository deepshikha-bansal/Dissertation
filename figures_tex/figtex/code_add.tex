float addition (float num1,float num2)
{
     int n1,n2;
     float temp_result;
  
     //Convert num1 and num2 to fixed point using the function convert
     //as given in section 4.2.3
     n1=convert(num1);
     n2=convert(num2);
  
     //Return the result of fixed point addition
     temp_result=n1+n2;
  
     return temp_result;
}
